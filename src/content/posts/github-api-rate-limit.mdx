      async function fetchWithCache(url: string, ttl = 600_000) {
        const cached = await idb.get(url);
        if (cached && Date.now() - cached.ts < ttl) {
          return cached.data;
        }
        const data = await fetch(url).then(r => r.json());
        await idb.set(url, { data, ts: Date.now() });
        return data;
      }
